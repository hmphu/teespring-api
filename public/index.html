Getting Started:

To use the API is really easy! Currently v.1 supports 1 endpoint. This will gather all information about a Teespring campaign.

HTTP Request

GET: http://teespring-api.cloudapp.net/v01/[campaign-name]

This will return the available information about a specific campaign.

Expected output:


{  
   "status":"ok",
   "tee":"campaign-name]",
   "url":"[campaign-url]",
   "images":{  
      "front":"[front-image-url",
      "back":"[back-image-url]"
   },
   "total_sold":"[amount-currently-sold",
   "goal_amount":"[goal-amount]",
   "goal_date":"[goal-end-date]",
   "details":"[campaign-details]"
}
GET: http://teespring-api.cloudapp.net/v01/search/[search query]

This will return an array of JSON objects for a given search query.

Expected output:


{
   hits: [ ],
   nbHits: 0,
   page: 0,
   nbPages: 0,
   query: "sample search query",
}
Each item in hits returns the following: 

{
  id: ,
  name: ,
  description: ,
  url: ,
  tippingpoint: ,
  ordered: ,
  front_text: ,
  back_text: ,
  endcost: ,
  enddate: ,
  amount_ordered: ,
  primary_pic_url: ,
  secondary_pic_url: ,
  user_id: ,
  active_to_i: ,
  tag_names: ,
  publicly_searchable: ,
  tags: [ ],
  objectID: ,
  highlightResult: {
    name: {
      value: ,
      matchLevel: ,
      matchedWords: [ ]
    },
    description: {
      value: ,
      matchLevel: ,
      matchedWords: [ ]
    },
    url: {
      value: ,
      matchLevel: ,
      matchedWords: [ ]
    },
    tag_names: {
      value: ,
      matchLevel: ,
      matchedWords: [ ]
    }
  }
}

If there's a problem, we'll also let you know


{  
   "status":"error",
   "message":"Looks like the campaign you requested may not exists..."
}

Here is a simple Node.JS example:


var request = require('request');
request('teespring-api.cloudapp.net/v01/tudev', function (error, response, body) {
  if (!error && response.statusCode == 200) {
    JSON.parse(body);
  }
})
